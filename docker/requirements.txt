# Dependency list for the kdr portable GPU image.
#
# Consumed by docker/Dockerfile (`pip install -r docker/requirements.txt`).
# docker/bootstrap.sh then force-reinstalls the Zyphra transformers fork
# (lucaspirola/transformers@zaya1-patches, based on transformers 4.57.1) over
# the version installed here — so the `transformers` pin below only needs to
# resolve a hub-<1.0-compatible base, not match the fork exactly.
#
# Note: torch is installed separately (Dockerfile uses the cu130 wheel index).
# We pin the version range here for reference; the install command differs.

# --- Deep learning core ---
torch==2.11.0
# Capped <5 so the image base stays compatible with huggingface_hub<1.0
# (transformers 5.x requires hub 1.x). The Zyphra fork bootstrap.sh installs
# is itself transformers 4.57.1-based.
transformers>=4.57,<5
accelerate>=1.0.0
safetensors>=0.4.5
tokenizers>=0.20.0
sentencepiece>=0.2.0
einops>=0.8.0
# NOTE: `kernels` was a moe_compress dependency (the finegrained-fp8 path for
# the Qwen FP8 KD teacher) and is intentionally NOT listed — kdr never imports
# it, and it forces huggingface_hub>=1.10, which breaks transformers 4.57.1's
# Hub round-trip: list_repo_templates 404s on a missing additional_chat_
# templates/ dir under hub 1.x (caught live in Phase 7.1).

# --- Hugging Face plumbing ---
# Capped <1.0: transformers 4.57.1 requires huggingface-hub<1.0, and only
# hub <1.0 raises the EntryNotFoundError that transformers catches when a
# recovered repo lacks an additional_chat_templates/ dir. hub 1.x raises an
# uncaught RemoteEntryNotFoundError -> load-back round-trip fails (Phase 7.1).
huggingface_hub>=0.34,<1.0
datasets>=3.0.0

# --- Numerics + solvers ---
numpy>=1.26.0
scipy>=1.11.0       # Hungarian assignment in REAM
ortools>=9.10       # SimpleMinCostFlow for Stage 2 v2 capacitated assignment

# --- Misc runtime ---
peft>=0.13.0
pyyaml>=6.0
pydantic>=2.5.0     # config validation in kdr. kdr's bootstrap.sh runs
                    # `pip install -e . --no-deps` against the cloned tree,
                    # so transitive runtime deps must be pre-installed here.
                    # See ../pyproject.toml for the same list.
lm-eval>=0.4.5
# NOTE: `trackio` removed — moe_compress experiment-tracking dep, never
# imported by kdr, and it forces huggingface_hub>=1.10 (see hub pin above).
nvidia-ml-py>=12.0
psutil>=5.9
bitsandbytes>=0.49.2
# tilelang is fla's alternative kernel backend that flash-linear-attention
# REQUIRES on Hopper (H100/H200) when Triton >= 3.4.0 — the default Triton
# backend for `chunk_bwd_dqkwg` produces incorrect gradients on Hopper, and
# fla itself raises a RuntimeError pointing at tilelang as the fix.
# Reproduced 2026-05-14 on H200 SXM5 (Datacrunch FIN-03), torch 2.11.0+cu130
# (which ships Triton 3.4.x). See fla GitHub issue #640.
tilelang>=0.1.7

# --- Fast-path attention kernels: handled per environment, NOT here ---
# flash-linear-attention and causal-conv1d build CUDA kernels and need to
# link against the host's torch + nvcc. They are installed by:
#   - max_quality/docker/Dockerfile  (with --no-build-isolation against the
#     cu128-pinned torch)
#   - max_quality/hf_jobs/entrypoint_ablations.py  (PEP 723 metadata with
#     a prebuilt-wheel URL, since the HF Jobs UV env has no nvcc)
# Listing them here would force pip to source-build under PEP 517 isolation,
# which spawns a venv that ignores constraint files and pulls a fresh torch
# from the default PyPI index — mismatching the cu128 base.

# --- Test only ---
pytest>=8.0
